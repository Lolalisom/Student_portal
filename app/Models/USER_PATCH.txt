In app/Models/User.php add inside the class:

public function createdAssignments() { return $this->hasMany(\App\Models\Assignment::class, 'created_by'); }
public function submissions() { return $this->hasMany(\App\Models\Submission::class, 'student_id'); }
public function results() { return $this->hasMany(\App\Models\Result::class, 'student_id'); }
public function messagesSent() { return $this->hasMany(\App\Models\Message::class, 'from_user_id'); }
public function messagesReceived() { return $this->hasMany(\App\Models\Message::class, 'to_user_id'); }

public function isAdmin(): bool { return $this->role === 'admin'; }
public function isStudent(): bool { return $this->role === 'student'; }
